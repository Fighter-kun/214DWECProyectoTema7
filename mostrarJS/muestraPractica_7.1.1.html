1. Crea una aplicación web con un botón “Cargar texto” y un campo de texto. En el campo de texto, escribiremos el nombre de un fichero de texto, y cuando pulsemos el botón cargaremos el contenido del archivo de texto en un contenedor que aparecerá en el centro de la pantalla. Dentro del contenedor pondremos un botón que cierre el cuadro de texto. El texto lo mostraremos modificando su formato, aplicándole color y cambiando la fuente. Los ficheros de texto los obtendremos del servidor donde tenemos alojada nuestra aplicación, para ello crea varios archivos .txt para las pruebas. Si el nombre indicado no se encuentra, mostraremos un error. Utiliza una promesa para realizar la tarea de obtención del fichero del servidor. 


                function cargarTexto() {
                  var fileName = document.getElementById("fileName").value;
                  var textContainer = document.getElementById("textContainer");
                  var textContent = document.getElementById("textContent");
    
                  fetch(`https://daw214.ieslossauces.es/214DWECProyectoTema7/tmp/${fileName}.json`)
                    .then(response => {
                      if (!response.ok) {
                        throw new Error("Archivo no encontrado");
                      }
                      return response.text();
                    })
                    .then(data => {
                      textContent.innerHTML = data;
                      textContainer.style.display = "block";
                    })
                    .catch(error => {
                      alert(error.message);
                    });
                }
    
                function cerrarContenedor() {
                  document.getElementById("textContainer").style.display = "none";
                }
